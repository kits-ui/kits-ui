import { reactive, ref, watch } from 'vue';
export default {
  data() {
    return {};
  },
  setup() {
    const authCode = ref('');
    const authCodeRef = ref(null);
    watch(authCode, async (n, o) => {
      console.log({ n, o });
    });
    const reset = () => {
      authCodeRef.value.createCodeHandle();
    };
    return {
      authCode,
      authCodeRef,
      reset,
    };
  },
};
</script>
